Prebid.js header bidding integration

Mix direct affiliate offers with programmatic display revenue via Prebid.js. The TrackMMP SDK exposes a hook so a Prebid auction can run in parallel with /api/v1/serve and the higher-eCPM result wins.

1) Set a floor eCPM on the placement

In the admin → Inventory → [site] → Placements → Edit:

  • Floor eCPM — minimum bid you'll accept (e.g. 1.50 means $1.50 CPM). Below this, the direct offer always wins.
  • Prebid config — JSON wrapper config (see below).

2) Add Prebid.js to your site

Build a Prebid.js bundle at prebid.org with the adapters you have accounts with (Magnite, PubMatic, OpenX, AppNexus, ix, etc.).

<script src="/path/to/prebid.js" async></script>

3) Wire the ApogeeMobi onServe hook

The SDK fires window.ApogeeMobi.onServe(data, el) after rendering when the placement has prebid_config set. Use that hook to run an auction and replace the rendered HTML if a bid beats the floor.

window.ApogeeMobi = window.ApogeeMobi || {};
window.ApogeeMobi.onServe = function (data, slotEl) {
  var p = data.placement;
  if (!p.prebid_config) return;
  var floor = parseFloat(p.floor_ecpm) || 0;

  var cfg;
  try { cfg = JSON.parse(p.prebid_config); } catch (e) { return; }
  // cfg.bids = [ { bidder: 'magnite', params: {…} }, … ]

  pbjs.que.push(function () {
    pbjs.addAdUnits([{
      code:  'apg-' + p.slug,
      sizes: cfg.sizes || [[300, 250], [728, 90]],
      bids:  cfg.bids || [],
    }]);
    pbjs.requestBids({
      timeout: 1200,
      adUnitCodes: ['apg-' + p.slug],
      bidsBackHandler: function () {
        var top = pbjs.getHighestCpmBids('apg-' + p.slug)[0];
        if (!top) return;
        var cpm = top.cpm;       // dollars per 1000 impressions
        if (cpm < floor) return; // direct offer wins
        // Replace the ApogeeMobi-rendered HTML with the winning ad
        slotEl.innerHTML = '';
        pbjs.renderAd(slotEl, top.adId);
      },
    });
  });
};

4) Example prebid_config JSON

{
  "sizes": [[300, 250], [728, 90]],
  "bids": [
    { "bidder": "magnite",  "params": { "accountId": 12345, "siteId": 67890, "zoneId": 111 } },
    { "bidder": "pubmatic", "params": { "publisherId": "12345", "adSlot": "abc/123@300x250" } },
    { "bidder": "openx",    "params": { "unit": "5555555", "delDomain": "yourname-d.openx.net" } }
  ]
}
Each adapter needs its own publisher account. Sign up directly with the SSP and they'll give you the params values. You usually start with 2-3 adapters and add more over time.

How the auction works

  1. SDK calls /api/v1/serve → direct offer comes back (Brand X, $250 CPA).
  2. SDK renders the direct offer immediately (no delay).
  3. onServe hook fires. Prebid runs its auction in parallel (1200ms timeout).
  4. If a programmatic bid > floor_ecpm, the rendered HTML is swapped out.
  5. Otherwise the direct offer stays. Best of both worlds.
This pattern means you never serve below your floor — direct affiliate revenue is the guaranteed floor; programmatic only wins when it beats your direct expected eCPM.
